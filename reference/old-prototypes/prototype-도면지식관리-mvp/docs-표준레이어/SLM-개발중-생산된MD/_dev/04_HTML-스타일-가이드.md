Markdown → HTML 변환 md = markdown.Markdown(extensions=[ 'tables', 'fenced_code', 'nl2br', ]) # 섹션별 분리 sections = _split_markdown_sections(markdown_text) html_sections = { 'risk_assessment_html': md.convert(sections.get('위험도 평가', '')), 'technical_analysis_html': md.convert(sections.get('기술 분석', '')), } # 2. 템플릿 로드 with open('config/global/templates/battery_report.html', 'r', encoding='utf-8') as f: template_str = f.read() template = Template(template_str) # 3. 데이터 병합 html = template.render( analysis_id=analysis_data['analysis_id'], timestamp=datetime.now().strftime('%Y-%m-%d %H:%M:%S'), critical_count=analysis_data['critical_count'], warning_count=analysis_data['warning_count'], normal_count=analysis_data['normal_count'], top_risks=analysis_data['top_risks'], recommendations=analysis_data['recommendations'], **html_sections ) return html def _split_markdown_sections(markdown_text: str) -> dict: """Markdown을 섹션별로 분리""" sections = {} current_section = None current_content = [] for line in markdown_text.split('\n'): if line.startswith('## '): if current_section: sections[current_section] = '\n'.join(current_content) current_section = line[3:].strip() current_content = [] else: current_content.append(line) if current_section: sections[current_section] = '\n'.join(current_content) return sections ``` --- ## 📊 차트 임베딩 (선택 사항) ### Chart.js 임베디드 ```html <!-- 차트 섹션 추가 (템플릿 내) --> <section class="section"> <h2 class="section-title">위험도 분포</h2> <canvas id="riskChart" width="400" height="200"></canvas> </section> <script> // Chart.js 라이브러리 임베디드 (CDN 대신 로컬) // (차트 라이브러리 코드 생략 - 실제로는 전체 코드 임베드) // 차트 생성 const ctx = document.getElementById('riskChart').getContext('2d');
