### 2.2 위험도 우선순위 계산

```python
# 위험도 점수 계산 로직
def calculate_risk_score(event, category):
    """위험도 점수 계산 (0-100)"""

    if category == "thermal":
        # dT/dt 기반 점수
        dtdt = event.get("dtdt", 0)
        threshold = 2.0  # Critical 임계값
        score = (dtdt / threshold) * 50

        # 온도 절대값 반영
        temp = event.get("temperature", 25)
        if temp > 55:
            score += 30
        elif temp > 45:
            score += 15

    elif category == "spc":
        # Z-Score 기반 점수
        z_score = abs(event.get("z_score", 0))
        threshold = 3.0  # Critical 임계값
        score = (z_score / threshold) * 50

    elif category == "correlation":
        # 상관계수 기반 점수
        corr = event.get("correlation", 1.0)
        threshold = 0.90  # 정상 임계값
        score = (1.0 - corr) / (1.0 - threshold) * 40

    return min(score, 100)  # 최대 100점

# 모든 이벤트에 위험도 점수 부여 및 정렬
all_risks = []
for category, data in integrated_data.items():
    for event in data.get("critical_events", []):
        event["risk_score"] = calculate_risk_score(event, category)
        event["category"] = category
        all_risks.append(event)

# 위험도 점수 기준 내림차순 정렬
all_risks.sort(key=lambda x: x["risk_score"], reverse=True)
```

### 2.4 Ollama AI 실행

```python
import requests

def call_ollama_ai(prompt, model="qwen2.5:32b", temperature=0.1, timeout=60):
    """Ollama AI 호출"""

    url = "http://localhost:11434/api/generate"

    payload = {
        "model": model,
        "prompt": prompt,
        "temperature": temperature,
        "stream": False
    }

    try:
        response = requests.post(url, json=payload, timeout=timeout)
        response.raise_for_status()

        result = response.json()
        insight_text = result.get("response", "")

        return insight_text

    except requests.exceptions.RequestException as e:
        print(f"[ERROR] Ollama AI 호출 실패: {e}")
        return None
```

### 2.5 후처리 및 검증

```python
import re

def postprocess_insight(insight_text):
    """AI 출력 후처리 (Markdown/이모지 제거)"""

    # 1. Markdown 헤더 제거 (# ## ### 등)
    insight_text = re.sub(r'^#{1,6}\s+', '', insight_text, flags=re.MULTILINE)

    # 2. 볼드 제거 (**텍스트**)
    insight_text = re.sub(r'\*\*(.+?)\*\*', r'\1', insight_text)

    # 3. 리스트 기호 제거 (- 또는 * 로 시작하는 리스트)
    insight_text = re.sub(r'^\s*[-*]\s+', '', insight_text, flags=re.MULTILINE)

    # 4. 이모지 제거 (Unicode emoji range)
    emoji_pattern = re.compile(
        "["
        "\U0001F600-\U0001F64F"  # emoticons
        "\U0001F300-\U0001F5FF"  # symbols & pictographs
        "\U0001F680-\U0001F6FF"  # transport & map symbols
        "\U0001F700-\U0001F77F"  # alchemical symbols
        "\U0001F780-\U0001F7FF"  # Geometric Shapes Extended
        "\U0001F800-\U0001F8FF"  # Supplemental Arrows-C
        "\U0001F900-\U0001F9FF"  # Supplemental Symbols and Pictographs
        "\U0001FA00-\U0001FA6F"  # Chess Symbols
        "\U0001FA70-\U0001FAFF"  # Symbols and Pictographs Extended-A
        "\U00002702-\U000027B0"  # Dingbats
        "]+",
        flags=re.UNICODE
    )
    insight_text = emoji_pattern.sub('', insight_text)

    return insight_text.strip()


def validate_insight(insight_text):
    """AI 출력 검증"""

    # 필수 섹션 확인
    required_sections = [
        "1. 전체 시스템 요약",
        "2. 위험도 우선순위 분석",
        "3. 종합 권장 조치사항",
        "4. 종합 평가"
    ]

    for section in required_sections:
        if section not in insight_text:
            print(f"[WARNING] 필수 섹션 누락: {section}")
            return False

    # 최소 길이 확인 (너무 짧으면 오류)
    if len(insight_text) < 500:
        print(f"[WARNING] 인사이트 길이 부족: {len(insight_text)} < 500")
        return False

    # Markdown 문법 잔존 여부 확인
    if re.search(r'^\s*#{1,6}\s+', insight_text, flags=re.MULTILINE):
        print(f"[WARNING] Markdown 헤더 잔존")
        return False

    return True
```

### 3.2 실제 운영 (Ollama AI)

**파일**: `GEM/scripts/phase3_generate_insights.py`

```python
# 실제 Phase 3에서는 Ollama AI 사용
def generate_comprehensive_insight(phase2_data):
    """Ollama AI를 통한 종합 인사이트 자동 생성"""

    # 1. 프롬프트 구성
    prompt = generate_comprehensive_prompt(phase2_data)

    # 2. Ollama AI 호출 (최대 3회 재시도)
    for attempt in range(3):
        insight_text = call_ollama_ai(prompt, temperature=0.1)

        if insight_text:
            # 3. 후처리
            insight_text = postprocess_insight(insight_text)

            # 4. 검증
            if validate_insight(insight_text):
                return insight_text

        print(f"[RETRY] Ollama AI 재시도 {attempt + 1}/3")
        time.sleep(2 ** attempt)  # 지수 백오프

    # 5. Fallback: 샘플 데이터 사용
    print("[FALLBACK] Ollama AI 실패, 샘플 데이터 사용")
    return get_fallback_comprehensive_insight(phase2_data)
```

**특징**:
- Ollama AI가 Phase 2 데이터를 읽고 자동 생성
- Temperature=0.1로 일관성 보장
- Retry/Fallback 메커니즘

**용도**:
- 실제 Phase 3 파이프라인
- 자동화된 리포트 생성

## 5. Fallback 전략

### 5.1 Ollama AI 실패 시나리오

1. **서버 미실행**: Ollama 서버가 실행되지 않음 (http://localhost:11434)
2. **Timeout**: 60초 내 응답 없음
3. **출력 품질 불량**: Markdown/이모지 포함, 필수 섹션 누락
4. **네트워크 오류**: requests.exceptions.RequestException

## 9. 성능 최적화

### 9.1 병렬 처리

```python
from concurrent.futures import ThreadPoolExecutor

# 4개 인사이트 병렬 생성
with ThreadPoolExecutor(max_workers=4) as executor:
    futures = {
        executor.submit(generate_thermal_insight, phase2_data): 'thermal',
        executor.submit(generate_spc_insight, phase2_data): 'spc',
        executor.submit(generate_integrity_insight, phase2_data): 'integrity',
        executor.submit(generate_correlation_insight, phase2_data): 'correlation'
    }

    insights = {}
    for future in futures:
        insight_type = futures[future]
        insights[insight_type] = future.result()

# 종합 인사이트는 4개 인사이트 완료 후 생성
insights['comprehensive'] = generate_comprehensive_insight(phase2_data, insights)
```

**효과**:
- 4개 인사이트 생성 시간 감소 (60초 → 15초)
- 종합 인사이트만 순차 실행 (4개 인사이트 결과 필요)

### 9.2 캐싱

```python
# 동일한 Phase 2 데이터는 캐싱
import hashlib
import json

def get_cache_key(phase2_data):
    """Phase 2 데이터 해시 생성"""
    data_str = json.dumps(phase2_data, sort_keys=True)
    return hashlib.sha256(data_str.encode()).hexdigest()

def generate_with_cache(phase2_data, cache_dir=".cache"):
    """캐시를 활용한 인사이트 생성"""
    cache_key = get_cache_key(phase2_data)
    cache_file = Path(cache_dir) / f"{cache_key}.md"

    # 캐시 존재하면 재사용
    if cache_file.exists():
        print(f"[CACHE HIT] {cache_key}")
        return cache_file.read_text(encoding='utf-8')

    # 캐시 없으면 생성 후 저장
    insight_text = call_ollama_ai(...)
    cache_file.parent.mkdir(exist_ok=True)
    cache_file.write_text(insight_text, encoding='utf-8')

    return insight_text
```
