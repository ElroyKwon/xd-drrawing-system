```python
# modules/storage/parquet_writer.py
import pyarrow as pa
import pyarrow.parquet as pq
import pandas as pd
from pathlib import Path

class OptimizedParquetWriter:
    """시계열 데이터 최적화된 Parquet 작성기"""

    # 배터리 데이터 스키마
    BATTERY_SCHEMA = pa.schema([
        ('timestamp', pa.timestamp('ms')),
        ('rack_id', pa.dictionary(pa.int16(), pa.string())),  # 반복 문자열
        ('v_avg', pa.float32()),
        ('v_min', pa.float32()),
        ('v_max', pa.float32()),
        ('t_avg', pa.float32()),
        ('t_max', pa.float32()),
        ('soc', pa.float32()),
    ])

    @staticmethod
    def write_optimized(
        df: pd.DataFrame,
        output_path: Path,
        compression: str = "zstd"
    ):
        """최적화된 Parquet 쓰기

        Args:
            df: 저장할 DataFrame
            output_path: 출력 경로
            compression: 압축 방식 (zstd 권장)

        최적화 설정:
        - ZSTD 압축: Snappy 대비 15% 더 압축, 속도 유사
        - Row Group 1M: 쿼리 성능 최적
        - Dictionary Encoding: rack_id 같은 반복 문자열
        - Statistics 활성화: 쿼리 필터링 가속
        """
        table = pa.Table.from_pandas(
            df,
            schema=OptimizedParquetWriter.BATTERY_SCHEMA
        )

        pq.write_table(
            table,
            output_path,
            compression=compression,
            compression_level=3,  # ZSTD level (1~22, 3은 속도/압축 균형)
            row_group_size=1_000_000,  # 1M rows per group
            data_page_size=1_048_576,  # 1MB (시계열 스캔 최적)
            use_dictionary=True,  # rack_id 등 반복 문자열 압축
            write_statistics=True,  # Min/Max 통계 (쿼리 스킵 가능)
            version='2.6',  # Parquet 최신 버전
        )


    @staticmethod
    def read_optimized(
        file_path: Path,
        columns: list = None,
        filters: list = None
    ) -> pd.DataFrame:
        """최적화된 Parquet 읽기

        Args:
            file_path: Parquet 파일 경로
            columns: 읽을 컬럼 (None이면 전체)
            filters: PyArrow 필터 (예: [('rack_id', '=', 'RACK_042')])

        Returns:
            Pandas DataFrame

        예시:
            # 특정 Rack만 읽기 (Row Group 스킵)
            df = read_optimized(
                'results/20251201/rack_data.parquet',
                columns=['timestamp', 'v_avg', 't_avg'],
                filters=[('rack_id', '=', 'RACK_042')]
            )
        """
        return pq.read_table(
            file_path,
            columns=columns,
            filters=filters,
            use_threads=True,  # 멀티스레드 읽기
        ).to_pandas()
```

```python
# modules/storage/streaming_writer.py
import pyarrow as pa
import pyarrow.parquet as pq

class StreamingParquetWriter:
    """메모리 초과 방지 스트리밍 작성"""

    def __init__(self, output_path: str, schema: pa.Schema):
        self.output_path = output_path
        self.writer = pq.ParquetWriter(
            output_path,
            schema,
            compression='zstd',
            compression_level=3
        )

    def write_batch(self, df_batch: pd.DataFrame):
        """배치 단위로 추가 (메모리 효율적)"""
        table = pa.Table.from_pandas(df_batch)
        self.writer.write_table(table)

    def close(self):
        """파일 닫기"""
        self.writer.close()


# === 사용 예시 ===
def save_large_dataset(data_iterator, output_path: str):
    """340M 행 데이터를 스트리밍으로 저장"""

    schema = OptimizedParquetWriter.BATTERY_SCHEMA
    writer = StreamingParquetWriter(output_path, schema)

    for batch_df in data_iterator:  # 1M 행씩 배치
        writer.write_batch(batch_df)

    writer.close()
```

```python
# modules/math_engine_polars.py
import polars as pl
import pandas as pd

class PolarsMathEngine:
    """Polars 기반 고성능 연산 엔진"""

    @staticmethod
    def calculate_zscore_polars(parquet_path: str) -> pl.DataFrame:
        """Z-Score 계산 (Polars - 3배 빠름)

        Args:
            parquet_path: Parquet 파일 경로

        Returns:
            Polars DataFrame with z_score column
        """
        df = pl.scan_parquet(parquet_path)  # Lazy 로딩

        result = df.select([
            pl.col('rack_id'),
            pl.col('timestamp'),
            pl.col('v_avg'),
            (
                (pl.col('v_avg') - pl.col('v_avg').mean()) /
                pl.col('v_avg').std()
            ).alias('z_score')
        ]).filter(
            pl.col('z_score').abs() > 3.0  # 임계값 초과만
        ).collect()  # 실행

        return result


    @staticmethod
    def calculate_dtdt_polars(parquet_path: str) -> pl.DataFrame:
        """온도 변화율 계산 (Polars - 스트리밍)

        Args:
            parquet_path: Parquet 파일 경로

        Returns:
            Polars DataFrame with dtdt column
        """
        df = pl.scan_parquet(parquet_path)

        result = df.sort('timestamp').select([
            pl.col('rack_id'),
            pl.col('timestamp'),
            pl.col('t_max'),
            (
                pl.col('t_max').diff() /
                pl.col('timestamp').diff().dt.total_seconds() * 60  # per minute
            ).alias('dtdt')
        ]).filter(
            pl.col('dtdt') > 2.0  # 임계값 초과만
        ).collect(streaming=True)  # 스트리밍 실행 (메모리 절약)

        return result


    @staticmethod
    def to_pandas(polars_df: pl.DataFrame) -> pd.DataFrame:
        """Polars → Pandas 변환 (필요시)"""
        return polars_df.to_pandas()
```

```python
# modules/math_engine.py (기존 Pandas 유지)
import pandas as pd
from modules.math_engine_polars import PolarsMathEngine

class MathEngine:
    """연산 엔진 (Pandas/Polars 혼용)"""

    def __init__(self, use_polars: bool = True):
        """
        Args:
            use_polars: True면 Polars 사용, False면 Pandas
        """
        self.use_polars = use_polars

    def calculate_zscore(self, data_path: str) -> pd.DataFrame:
        """Z-Score 계산 (자동 선택)"""
        if self.use_polars:
            polars_result = PolarsMathEngine.calculate_zscore_polars(data_path)
            return polars_result.to_pandas()
        else:
            # 기존 Pandas 코드
            df = pd.read_parquet(data_path)
            df['z_score'] = (df['v_avg'] - df['v_avg'].mean()) / df['v_avg'].std()
            return df[df['z_score'].abs() > 3.0]
```

```javascript
// static/viewer/v1/websocket_manager.js

class WebSocketManager {
    /**
     * WebSocket 연결 관리 (Exponential Backoff + Jitter)
     *
     * 특징:
     * - 자동 재연결 (최대 10회)
     * - Exponential Backoff (1s → 2s → 4s → ... → 30s)
     * - Jitter 추가 (동시 재연결 방지)
     */
    constructor(url, onMessage, onError) {
        this.url = url;
        this.onMessage = onMessage;
        this.onError = onError;

        // 재연결 설정
        this.reconnectDelay = 1000;  // 초기 1초
        this.maxDelay = 30000;        // 최대 30초
        this.reconnectAttempts = 0;
        this.maxAttempts = 10;

        // 연결 상태
        this.ws = null;
        this.isIntentionallyClosed = false;

        this.connect();
    }

    connect() {
        if (this.isIntentionallyClosed) return;

        console.log(`[WS] Connecting to ${this.url}...`);
        this.ws = new WebSocket(this.url);

        // 연결 성공
        this.ws.onopen = () => {
            console.log('[WS] Connected');
            this.reconnectAttempts = 0;  // 리셋
            this.reconnectDelay = 1000;
        };

        // 메시지 수신
        this.ws.onmessage = (event) => {
            try {
                const data = JSON.parse(event.data);
                this.onMessage(data);
            } catch (e) {
                console.error('[WS] Parse error:', e);
            }
        };

        // 연결 종료
        this.ws.onclose = (event) => {
            if (this.isIntentionallyClosed) {
                console.log('[WS] Closed intentionally');
                return;
            }

            console.warn(`[WS] Disconnected (code: ${event.code})`);

            if (this.reconnectAttempts < this.maxAttempts) {
                this.scheduleReconnect();
            } else {
                console.error('[WS] Max reconnect attempts reached');
                this.onError('재연결 실패 (최대 시도 초과)');
            }
        };

        // 에러 발생
        this.ws.onerror = (error) => {
            console.error('[WS] Error:', error);
        };
    }

    scheduleReconnect() {
        // Exponential Backoff + Jitter
        const delay = Math.min(
            this.reconnectDelay * Math.pow(2, this.reconnectAttempts),
            this.maxDelay
        ) + Math.random() * 1000;  // 0~1초 Jitter

        console.log(`[WS] Reconnecting in ${(delay / 1000).toFixed(1)}s...`);

        setTimeout(() => {
            this.reconnectAttempts++;
            this.connect();
        }, delay);
    }

    send(data) {
        if (this.ws && this.ws.readyState === WebSocket.OPEN) {
            this.ws.send(JSON.stringify(data));
        } else {
            console.warn('[WS] Not connected, cannot send');
        }
    }

    close() {
        this.isIntentionallyClosed = true;
        if (this.ws) {
            this.ws.close();
        }
    }
}

// === 사용 예시 ===
const wsManager = new WebSocketManager(
    'ws://localhost:8000/ws/analysis/ABC123',
    (data) => {
        // 메시지 수신 처리
        console.log('Received:', data);
        updateProgressBar(data.progress);
    },
    (error) => {
        // 에러 처리
        alert(error);
    }
);
```

```python
from fastapi import WebSocket, WebSocketDisconnect
import asyncio

class ConnectionManager:
    """WebSocket 연결 관리"""

    def __init__(self):
        self.active_connections: dict[str, WebSocket] = {}

    async def connect(self, trace_id: str, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[trace_id] = websocket

    def disconnect(self, trace_id: str):
        self.active_connections.pop(trace_id, None)

    async def send_message(self, trace_id: str, message: dict):
        websocket = self.active_connections.get(trace_id)
        if websocket:
            try:
                await websocket.send_json(message)
            except:
                self.disconnect(trace_id)

    async def heartbeat(self, trace_id: str):
        """30초마다 Ping 전송 (연결 유지)"""
        while trace_id in self.active_connections:
            await asyncio.sleep(30)
            await self.send_message(trace_id, {"type": "heartbeat"})


manager = ConnectionManager()

@app.websocket("/ws/analysis/{trace_id}")
async def websocket_endpoint(websocket: WebSocket, trace_id: str):

    # Heartbeat 시작
    heartbeat_task = asyncio.create_task(manager.heartbeat(trace_id))

            data = await websocket.receive_text()
            # ...
    except WebSocketDisconnect:
        manager.disconnect(trace_id)
```

```python
import os
from concurrent.futures import ProcessPoolExecutor, as_completed
from typing import List, Callable
import logging

logger = logging.getLogger(__name__)


class OptimizedWorkerPool:
    """최적화된 CPU 병렬 처리"""

    def __init__(self):
        # CPU 코어 수 감지
        cpu_count = os.cpu_count() or 4

        # 워커 수 계산 (1 코어는 OS용 예약)
        # 160 Rack / 8 Core = 20 Rack per worker
        self.max_workers = min(cpu_count - 1, 8)

        logger.info(f"🔧 Worker Pool: {self.max_workers} workers")

    def parallel_execute(
        self,
        task_func: Callable,
        task_args_list: List[tuple],
        timeout: int = 300
    ) -> List:
        """병렬 실행

        Args:
            task_func: 실행할 함수
            task_args_list: 각 태스크의 인자 리스트 [(rack_id, data), ...]
            timeout: 태스크 타임아웃 (초)

        Returns:
            결과 리스트
        """
        results = []

        with ProcessPoolExecutor(
            max_workers=self.max_workers,
            initializer=self._init_worker  # 워커 초기화 1회만
        ) as executor:

            # 모든 태스크 제출
            future_to_args = {
                executor.submit(task_func, *args): args
                for args in task_args_list
            }

            # 완료된 순서대로 수집
            for future in as_completed(future_to_args, timeout=timeout):
                args = future_to_args[future]
                try:
                    result = future.result()
                    results.append(result)
                except Exception as e:
                    logger.error(f"Task failed for {args[0]}: {e}")

        - 로거 설정
        """
```

```python
# modules/hot_path/threshold_checker.py
import polars as pl
from typing import List, Dict

class ThresholdChecker:
    """실시간 임계값 체크 (Hot Path)

    목표: < 1초 SLA
    방법: 복잡한 연산 없이 단순 비교만
    """

    # 임계값 정의
    THRESHOLDS = {
        'dtdt': 2.0,        # °C/min
        'z_score': 3.0,     # σ
        't_max': 60.0,      # °C
        'v_min': 2.5,       # V
    }

    @staticmethod
    def check_critical_risks(parquet_path: str) -> List[Dict]:
        """긴급 위험 감지 (Polars 스트리밍)

        Args:
            parquet_path: 실시간 데이터 Parquet

        Returns:
            긴급 위험 리스트 [{'rack_id': 'RACK_042', 'risk': 'THERMAL', ...}]
        """
        df = pl.scan_parquet(parquet_path)

        # 1. 온도 변화율 (dT/dt)
        thermal_risks = df.sort('timestamp').select([
            pl.col('rack_id'),
            pl.col('t_max'),
            (
                pl.col('t_max').diff() /
                pl.col('timestamp').diff().dt.total_seconds() * 60
            ).alias('dtdt')
        ]).filter(
            pl.col('dtdt') > ThresholdChecker.THRESHOLDS['dtdt']
        ).collect(streaming=True)

        # 2. 절대 온도 초과
        temp_risks = df.filter(
            pl.col('t_max') > ThresholdChecker.THRESHOLDS['t_max']
        ).select(['rack_id', 't_max']).collect(streaming=True)

        # 3. 전압 하한 이탈
        voltage_risks = df.filter(
            pl.col('v_min') < ThresholdChecker.THRESHOLDS['v_min']
        ).select(['rack_id', 'v_min']).collect(streaming=True)

        # 결과 통합
        critical_risks = []

        for row in thermal_risks.iter_rows(named=True):
            critical_risks.append({
                'rack_id': row['rack_id'],
                'risk_type': 'THERMAL',
                'value': row['dtdt'],
                'threshold': ThresholdChecker.THRESHOLDS['dtdt']
            })

        for row in temp_risks.iter_rows(named=True):
            critical_risks.append({
                'rack_id': row['rack_id'],
                'risk_type': 'TEMP_HIGH',
                'value': row['t_max'],
                'threshold': ThresholdChecker.THRESHOLDS['t_max']
            })

        return critical_risks


# === WebSocket 즉시 알림 ===
async def instant_alert(trace_id: str, critical_risks: List[Dict]):

            'risks': critical_risks,

        })
```
