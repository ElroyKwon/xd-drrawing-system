## 🚀 실행 방법

### 개발 모드 (Dev Console)
```bash
# 1. Ollama 서버 시작
ollama serve

# 2. Qwen2.5 32B 모델 다운로드 (4-bit 양자화 - 24GB VRAM)
ollama pull qwen2.5:32b-instruct-q4_K_M

# 3. 모델 확인
ollama list

# 4. Dev Console 실행
python dev_server.py

# 브라우저에서
http://localhost:8000/admin
```

### 운영 모드 (Production with Ollama - 권장)
```bash
# 1. 환경변수 설정
export OLLAMA_HOST=http://localhost:11434
export $(cat .env | xargs)

# 2. Ollama 서비스로 등록 (Windows)
# NSSM으로 Ollama를 서비스로 실행
nssm install OllamaService "C:\Program Files\Ollama\ollama.exe" "serve"
nssm start OllamaService

# 3. API 서버 실행
uvicorn main:app --host 0.0.0.0 --port 8000 --workers 4

# 또는 Gunicorn (Unix 계열)
gunicorn main:app -w 4 -k uvicorn.workers.UvicornWorker
```

### 대규모 운영 모드 (vLLM - 선택 사항)
```bash
# ⚠️ 동시 요청 100+ 시에만 사용

# 1. vLLM 서버 시작 (A100 2개 이상 필요)
docker run --gpus all \
  -v ~/.cache/huggingface:/root/.cache/huggingface \
  -p 8001:8000 \
  vllm/vllm-openai:latest \
  --model Qwen/Qwen2.5-32B-Instruct \
  --tensor-parallel-size 2 \
  --dtype bfloat16

# 2. 환경변수에서 vLLM 활성화
export USE_VLLM=true
export VLLM_HOST=http://localhost:8001

# 3. API 서버 실행 (동일)
uvicorn main:app --host 0.0.0.0 --port 8000 --workers 4
```

```python
import logging
from logging.handlers import RotatingFileHandler

logger = logging.getLogger("GEM")
logger.setLevel(logging.INFO)

# 파일 핸들러 (10MB, 5개 백업)
file_handler = RotatingFileHandler(
    "logs/gem.log",
    maxBytes=10*1024*1024,
    backupCount=5
)

# 포맷
formatter = logging.Formatter(
    '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
file_handler.setFormatter(formatter)
logger.addHandler(file_handler)
```

### 성능 측정
```python
# modules/performance_monitor.py
import time
import functools

def measure_time(phase: str, threshold: int = None):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            start = time.perf_counter()
            result = func(*args, **kwargs)
            elapsed_ms = (time.perf_counter() - start) * 1000

            logger.info(f"{phase}: {elapsed_ms:.2f}ms")

            if threshold and elapsed_ms > threshold:
                logger.warning(f"{phase} SLA 초과: {elapsed_ms:.2f}ms > {threshold}ms")

            return result
        return wrapper
    return decorator

# 사용 예시
@measure_time(phase="Phase 2", threshold=10000)
def execute_math_engine():
    # ...
```
