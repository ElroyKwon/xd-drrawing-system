### 2. 설정 로더 (`config/settings.py`)

```python
# config/settings.py
from pydantic_settings import BaseSettings
from typing import Optional

class Settings(BaseSettings):
    """GEM 프로젝트 설정"""

    # 보안
    gem_api_key: str

    # SLM 모델
    slm_model_primary: str = "qwen2.5:32b-instruct-q4_K_M"
    slm_model_fallback: str = "gemma2:27b"
    slm_model_active: str = "qwen2.5:32b-instruct-q4_K_M"

    # Inference Engine
    use_ollama: bool = True
    ollama_host: str = "http://localhost:11434"
    use_vllm: bool = False
    vllm_host: str = "http://localhost:8001"

    # 성능
    use_polars: bool = True
    max_workers: int = 8

    # 로그
    log_level: str = "INFO"

    class Config:
        env_file = ".env"
        env_file_encoding = "utf-8"
        case_sensitive = False  # 대소문자 구분 안 함


# 싱글톤 인스턴스
settings = Settings()


def get_model_config() -> dict:
    """현재 모델 설정 반환"""

    if settings.use_vllm and settings.vllm_host:
        return {
            "model_name": settings.slm_model_active,
            "api_base": f"{settings.vllm_host}/v1",
            "engine": "vLLM"
        }
    else:
        return {
            "model_name": settings.slm_model_active,
            "api_base": f"{settings.ollama_host}/v1",
            "engine": "Ollama"
        }
```

### 3. SLM 클라이언트 수정 (`modules/intelligence/llm_client.py`)

```python
# modules/intelligence/llm_client.py
import dspy
from config.settings import settings, get_model_config
from modules.intelligence.dspy_prompts import BatteryAnalyzer
from modules.intelligence.guardrails import GuardrailValidator
import logging

logger = logging.getLogger(__name__)


class SLMClient:
    """SLM 클라이언트 (환경변수 기반)"""

    def __init__(self, use_optimized: bool = False):
        """
        Args:
            use_optimized: 최적화된 프롬프트 사용 여부

        환경변수에서 자동으로 모델/엔진 선택:
        - SLM_MODEL_ACTIVE: 사용할 모델
        - USE_OLLAMA / USE_VLLM: 엔진 선택
        """
        # 환경변수에서 설정 로드
        config = get_model_config()

        logger.info(f"🤖 SLM 초기화: {config['engine']} - {config['model_name']}")

        # DSPy 설정
        lm = dspy.OpenAI(
            model=config['model_name'],
            api_base=config['api_base'],
            api_key="ollama"  # Ollama는 더미 키
        )
        dspy.settings.configure(lm=lm)

        # Analyzer 로드
        if use_optimized:
            self.analyzer = BatteryAnalyzer()
            try:
                self.analyzer.load("models/optimized_battery_analyzer.json")
                logger.info("✅ 최적화된 프롬프트 로드")
            except FileNotFoundError:
                logger.warning("⚠️ 최적화 파일 없음, 기본 프롬프트 사용")
        else:
            self.analyzer = BatteryAnalyzer()

        # Guardrail 설정
        self.validator = GuardrailValidator(timeout=60)

    async def analyze_battery_risks(
        self,
        summary_data: str,
        physics_validation: str
    ) -> str:
        """배터리 위험 분석

        Args:
            summary_data: YAML 요약 데이터
            physics_validation: 물리 검증 결과

        Returns:
            Markdown 리포트
        """
        return await self.validator.validate_and_execute(
            self.analyzer,
            summary_data,
            physics_validation
        )
```

## 🛠️ Ollama 모델 관리

### 자동 설치 문제 해결

**Q: Ollama를 자동으로 설치하면 나중에 사양이 바뀌면 옵션을 다시 설치해야 하나?**

**A: 아니요! 모델은 독립적으로 관리됩니다.**

### Ollama 모델 명령어

```bash
# === 모델 다운로드 ===
# 4-bit 양자화 (권장)
ollama pull qwen2.5:32b-instruct-q4_K_M

# 8-bit 양자화 (고정확도)
ollama pull qwen2.5:32b-instruct-q8_0

# === 설치된 모델 확인 ===
ollama list

# 출력 예시:
# NAME                             SIZE      MODIFIED
# qwen2.5:32b-instruct-q4_K_M     20GB      2 hours ago
# gemma2:27b                       16GB      1 day ago

# === 모델 삭제 (공간 확보) ===
ollama rm gemma2:27b

# === 모델 전환 (자동) ===
# .env만 수정하면 Ollama가 자동으로 해당 모델 사용
# 다운로드 안 되어 있으면 자동 다운로드 (Ollama >= 0.1.26)

# === 수동 테스트 ===
ollama run qwen2.5:32b-instruct-q4_K_M
>>> RACK_042에서 dT/dt 5.2도/분 감지. 위험도는?
```

### 모델 저장 위치

```bash
# Windows
C:\Users\{사용자}\.ollama\models\

# Linux
~/.ollama/models/

# 모델 파일 크기
qwen2.5:32b-instruct-q4_K_M  → 20GB
qwen2.5:32b-instruct-q8_0    → 32GB
gemma2:27b                    → 16GB
```

**09_SLM-프롬프트-전략.md 433줄:**
```python
# ❌ 잘못된 설정
lm = dspy.OpenAI(

    api_base="http://localhost:11434/v1",
```

### ✅ 올바른 설정

```python
# config/settings.py 기반 자동 선택
from config.settings import get_model_config

config = get_model_config()

# Ollama 사용 시
if config['engine'] == 'Ollama':
    lm = dspy.OpenAI(
        model="qwen2.5:32b-instruct-q4_K_M",  # Ollama 형식
        api_base="http://localhost:11434/v1",
        api_key="ollama"  # 더미 키
    )

# vLLM 사용 시
elif config['engine'] == 'vLLM':
    lm = dspy.OpenAI(
        model="Qwen/Qwen2.5-32B-Instruct",  # HuggingFace 형식
        api_base="http://localhost:8001/v1",
        api_key="EMPTY"  # vLLM은 EMPTY 사용
    )
```
